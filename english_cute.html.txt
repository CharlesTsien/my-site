        const levels = [
            { id: 'level1', label: '🌱 级别一：认读 + 拼写', unlocked: true },
            { id: 'level2', label: '🌿 级别二：词性大挑战', unlocked: l1 === 'true' },
            { id: 'level3', label: '🌳 级别三：中文释义对战', unlocked: l2 === 'true' }
        ];

        const grid = document.getElementById('level-grid');
        grid.innerHTML = '';
        levels.forEach(lv => {
            const div = document.createElement('div');
            div.className = 'unit-card';
            if(!lv.unlocked) div.classList.add('locked');
            div.innerHTML = `<h3>${lv.label}</h3>`;
            if(lv.unlocked) {
                div.onclick = () => {
                    state.currentLevel = lv.id;
                    startGame();
                };
            } else {
                div.innerHTML += `<p style="font-size:0.8rem; color:#ff6b6b;">🔒 需要完成上一级别</p>`;
            }
            grid.appendChild(div);
        });
    }

    function startGame() {
        showScreen('screen-game');
        const unit = db.units[state.currentUnit];
        state.sessionWords = shuffle([...unit.words]);
        state.currentWordIndex = 0;
        state.score = 0;
        state.total = 0;
        state.isGameOver = false;
        document.getElementById('btn-unmastered').style.display = 'block';
        document.getElementById('game-result').style.display = 'none';
        renderQuestion();
    }

    function renderQuestion() {
        if(state.currentWordIndex >= state.sessionWords.length) {
            finishGame();
            return;
        }

        const wordObj = state.sessionWords[state.currentWordIndex];
        document.getElementById('game-progress-text').textContent = `${state.currentWordIndex+1} / ${state.sessionWords.length}`;
        document.getElementById('game-progress-fill').style.width = `${((state.currentWordIndex+1) / state.sessionWords.length) * 100}%`;
        document.getElementById('game-score-text').textContent = `✅ ${state.score}`;

        const content = document.getElementById('game-content');
        const heading = document.getElementById('game-heading');
        const instruction = document.getElementById('game-instruction');

        let html = `<div style="font-size: 2rem; text-align: center; margin: 20px 0; background: #f0f8ff; padding: 20px; border-radius: 30px;">${wordObj.word}</div>`;

        if(state.currentLevel === 'level1') {
            heading.textContent = '🎧 听单词，拼出来';
            instruction.textContent = '点击喇叭听发音，然后输入拼写。';
            html += `
                <div style="text-align: center; margin-top: -15px;">
                    <button class="btn btn-small btn-secondary" onclick="speakText('${wordObj.word}')" style="width: auto; background: #87ceeb; box-shadow: 0 4px 0 #4682b4;">🔊 播放读音</button>
                </div>
                <div class="input-area">
                    <input type="text" id="game-input" placeholder="输入英文单词..." autofocus>
                </div>
            `;
            content.innerHTML = html;

        } else if(state.currentLevel === 'level2') {
            heading.textContent = '📝 请选择正确的词性';
            instruction.textContent = `【 ${wordObj.word} 】的词性是？`;
            const correct = wordObj.pos || '短语';
            let options = [correct];
            // 找不同的干扰项
            const availablePOS = POS_OPTIONS.filter(p => p !== correct);
            while(options.length < 4) {
                let rand = availablePOS[Math.floor(Math.random() * availablePOS.length)];
                if(!options.includes(rand)) options.push(rand);
            }
            options = shuffle(options);
            html += `
                <div class="options-grid" id="choice-grid">
                    ${options.map((o, idx) => `<div class="option-btn" data-pos="${o}" onclick="selectOption(this, '${o}', '${correct}')">${o}</div>`).join('')}
                </div>
            `;
            content.innerHTML = html;

        } else if(state.currentLevel === 'level3') {
            heading.textContent = '📖 这个单词是什么意思？';
            instruction.textContent = `【 ${wordObj.word} 】的中文含义是？`;
            const correct = wordObj.meaning;
            // 找不同的干扰项
            let options = [correct];
            let pool = db.units[state.currentUnit].words.map(w => w.meaning);
            while(options.length < 4) {
                let rand = pool[Math.floor(Math.random() * pool.length)];
                if(!options.includes(rand)) options.push(rand);
            }
            options = shuffle(options);
            html += `
                <div class="options-grid" id="choice-grid">
                    ${options.map((o, idx) => `<div class="option-btn" data-meaning="${o}" onclick="selectOption(this, '${o}', '${correct}')">${o}</div>`).join('')}
                </div>
            `;
            content.innerHTML = html;
        }
    }

    function selectOption(el, selected, correct) {
        if(el.classList.contains('correct') || state.isGameOver) return;
        const btns = document.querySelectorAll('.option-btn');
        btns.forEach(b => b.style.pointerEvents = 'none');
        
        if(selected === correct) {
            el.classList.add('correct');
            state.score++;
            state.total++;
            setTimeout(nextQuestion, 800);
        } else {
            el.classList.add('wrong');
            btns.forEach(b => { if(b.textContent === correct) b.classList.add('correct'); });
            state.total++;
            setTimeout(nextQuestion, 1500);
        }
    }

    function submitAnswer() {
        if(state.currentLevel === 'level1' && !state.isGameOver) {
            const input = document.getElementById('game-input');
            if(!input || !input.value) return alert("✨ 先输入拼写再提交哦！");
            const wordObj = state.sessionWords[state.currentWordIndex];
            const ans = input.value.trim().toLowerCase();
            const correct = wordObj.word.toLowerCase();
            
            if(ans === correct) {
                state.score++;
                state.total++;
                input.style.borderColor = '#98fb98';
                setTimeout(nextQuestion, 800);
            } else {
                state.total++;
                input.style.borderColor = '#ff6b6b';
                setTimeout(nextQuestion, 1500);
            }
        }
    }

    function nextQuestion() {
        state.currentWordIndex++;
        renderQuestion();
    }

    function finishGame() {
        state.isGameOver = true;
        document.getElementById('btn-unmastered').style.display = 'none';
        const progress = document.getElementById('game-progress-fill');
        progress.style.width = '100%';
        document.getElementById('game-progress-text').textContent = `完成！`;

        const unitStr = `unit_${state.currentUnit}_${state.currentLevel}_done`;
        localStorage.setItem(unitStr, 'true');

        const content = document.getElementById('game-content');
        const heading = document.getElementById('game-heading');
        heading.textContent = '🎉 闯关成功！';
        
        let html = `
            <div style="text-align: center; background: #f0f8ff; border-radius: 30px; padding: 20px;">
                <span style="font-size: 3rem;">🌻</span>
                <p style="font-size: 1.2rem; font-weight: bold;">答对 ${state.score} 题，共 ${state.total} 题</p>
            </div>
            <div class="btn-gold btn" onclick="showScreen('screen-levels')" style="width: auto; margin-top: 20px;">🔄 返回级别选择</div>
        `;
        content.innerHTML = html;
    }

    // 未掌握机制：艾宾浩斯复习清单
    function markUnmastered() {
        if(state.isGameOver || state.currentWordIndex >= state.sessionWords.length) return;
        const wordObj = state.sessionWords[state.currentWordIndex];
        if(!state.unmasteredList.find(w => w.word === wordObj.word)) {
            state.unmasteredList.push(wordObj);
            localStorage.setItem('cute_unmastered', JSON.stringify(state.unmasteredList));
        }
        alert(`📕 已将 "${wordObj.word}" 加入“未掌握”复习清单！系统将根据艾宾浩斯曲线提醒您。`);
        state.currentWordIndex++;
        renderQuestion();
    }

    function startReview() {
        if(state.unmasteredList.length === 0) {
            alert("🎉 恭喜！你的“未掌握”清单目前是空的！");
            return;
        }
        state.currentLevel = 'level1';
        state.sessionWords = shuffle([...state.unmasteredList]);
        state.currentWordIndex = 0;
        state.score = 0;
        state.total = 0;
        state.isGameOver = false;
        showScreen('screen-game');
        document.getElementById('btn-unmastered').style.display = 'none';
        document.getElementById('review-notification').innerHTML = `📖 复习模式进行中，共 ${state.sessionWords.length} 个未掌握单词。答对会自动移出清单。`;
        document.getElementById('review-notification').style.display = 'block';
        renderQuestion();
    }

    // 自然拼读 API (TTS)
    function speakText(text) {
        if(!window.speechSynthesis) return alert("⚠️ 您的浏览器不支持语音合成");
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = 'en-US';
        utterance.rate = 0.9;
        window.speechSynthesis.speak(utterance);
    }

    function shuffle(array) {
        let currentIndex = array.length, randomIndex;
        while (currentIndex != 0) {
            randomIndex = Math.floor(Math.random() * currentIndex);
            currentIndex--;
            [array[currentIndex], array[randomIndex]] = [array[randomIndex], array[currentIndex]];
        }
        return array;
    }

    // 监听回车键
    document.addEventListener('keydown', function(event) {
        if (event.key === 'Enter') {
            const input = document.getElementById('game-input');
            if(input && document.getElementById('screen-game').classList.contains('active-screen')) {
                submitAnswer();
            }
        }
    });

    initUnits();
